Сервисы располагабтся в папке services внутри модуля. Они представляют собой законченный функционал внутри одного класса.
Создаются специально для удобного использотвания другими программистами без необходимости изучать код.

В контроллере:
<?
/**
 * 1. Вызвали экземпляр
 */
$tSL = new TestService;
/**
 * 2. Задали параметры
 */
$params = [
	'count'=>7,
];
/**
 * 3. Отправили их в Service
 */
$tSL->load($params);
/**
 * 4. Получили результаты
 */
$results = $tSL->getResults();
/**
 * 5. Пихнули это во view
 */
$this->render('testView.php',[
	'results'=>$results,
]);
?>

Сам Service
<?

/**
 * Сам Service класть в {папка_компонента}/services/
 */
class TestService extends Service {
	/**
	 * Процесс возврата данных по полученным параметрам.
	 */
	protected $count;
	protected function loadResults() {
		/**
		 * Все параметры, которые были отправлены в контроллере, тут будут в $this->{param_name}
		 * Все результирующие данные должны быть засунуты в $this->results
		 */
		/**
		 * Список действий:
		 * 1. Получили параметры
		 * 2. Обработали их
		 * 3. Засунули результаты в $this->results
		 */
		$this->results = $this->count*3;

	}

	/**
	 * Процесс выяснения общего количества результатов поиска без учета лимитов и запись числа в аттрибут total_count
	 */
	protected function loadTotalCount() {
		/**
		 * Тут логика, которая вернет количество всех записей для пагинаторов в переменную total_count
		 */
		/**
		 * Список действий:
		 * 1. Получили количество записей
		 * 2. Засунули результат в $this->total_count
		 */
		$this->total_count = 1;
	}
}
